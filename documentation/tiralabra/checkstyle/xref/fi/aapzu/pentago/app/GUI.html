67      private void loadStartMenu() throws IOException {
68          replaceSceneContent("fxml/StartMenu.fxml");
69          TextField blackPlayerName = (TextField) (baseScene.getRoot().lookup("#whitePlayerName"));
70          TextField whitePlayerName = (TextField) (baseScene.getRoot().lookup("#blackPlayerName"));
71          Button startButton = (Button) (baseScene.getRoot().lookup("#startButton"));
72  
73          player0isBot = (CheckBox) (baseScene.getRoot().lookup("#player0isBot"));
74          player1isBot = (CheckBox) (baseScene.getRoot().lookup("#player1isBot"));
75  
76          EnterClickEventHandler handler = new EnterClickEventHandler() {
77              @Override
78              public void handleEvent() {
79                  try {
80                      String p1 = !blackPlayerName.getText().equals("") ? blackPlayerName.getText() : "Black";
81                      String p2 = !whitePlayerName.getText().equals("") ? whitePlayerName.getText() : "White";
82                      game.clearPlayers();
83                      if (player0isBot.isSelected()) {
84                          game.addBot(p1);
85                      } else {
86                          game.addHumanPlayer(p1);
87                      }
88                      if (player1isBot.isSelected()) {
89                          game.addBot(p2);
90                      } else {
91                          game.addHumanPlayer(p2);
92                      }
93                      loadGame();
94                  } catch (IOException ex) {
95                      Logger.getLogger(GUI.class.getName()).log(Level.SEVERE, null, ex);
96                  }
97              }
98          };
99          blackPlayerName.addEventHandler(KeyEvent.KEY_PRESSED, handler);
100         whitePlayerName.addEventHandler(KeyEvent.KEY_PRESSED, handler);
101         startButton.addEventHandler(MouseEvent.MOUSE_CLICKED, handler);
102     }

104     private void loadGame() throws IOException {
105         replaceSceneContent("fxml/Game.fxml");
106 
107         rotateButtons = new ArrayList<>();
108         rotateButtons.addAll(baseScene.getRoot().lookupAll(".counterClockwise"));
109 
110         helpLabel = (Label) (baseScene.getRoot().lookup("#helpLabel"));
111         errorLabel = (Label) (baseScene.getRoot().lookup("#errorLabel"));
112 
113         lockPane = (Pane) (baseScene.getRoot().lookup("#lockPane"));
114 
115         unlockBoardForPlayer();
116 
117         initTiles();
118         initCircles();
119         addCircleClickHandlers();
120 
121         nextTurn();
122     }

124     private void initTiles() {
125         Set<Node> tileSet = baseScene.getRoot().lookupAll(".tile");
126         int i = 0;
127         for (Node n : tileSet) {
128             final int x = i % 2;
129             final int y = i / 2;
130             tiles[y][x] = (GridPane) (n);
131             Node clockwise = n.lookup("#clockwise" + x + y);
132             Node counterClockwise = n.lookup("#counterClockwise" + x + y);
133             if (clockwise != null && counterClockwise != null) {
134                 rotateButtons.add(clockwise);
135                 rotateButtons.add(counterClockwise);
136                 clockwise.addEventHandler(MouseEvent.MOUSE_CLICKED, (MouseEvent event) -> {
137                     if (game.getAllowedToRotate()) {
138                         rotateTile(x, y, Direction.CLOCKWISE);
139                     }
140                 });
141                 counterClockwise.addEventHandler(MouseEvent.MOUSE_CLICKED, (MouseEvent event) -> {
142                     if (game.getAllowedToRotate()) {
143                         rotateTile(x, y, Direction.COUNTER_CLOCKWISE);
144                     }
145                 });
146             }
147             i++;
148         }
149     }
150 
151     private void initCircles() {
152         for (int y = 0; y < 2; y++) {
153             for (int x = 0; x < 2; x++) {
154                 GridPane tile = tiles[y][x];
155                 Set<Node> circleSet = tile.lookupAll("Circle");
156                 int i = 0;
157                 for (Node n : circleSet) {
158                     int cX = x * 3 + i % 3;
159                     int cY = y * 3 + i / 3;
160                     circles[cY][cX] = (Circle) n;
161                     i++;
162                 }
163             }
164         }
165     }
166 
167     private void addCircleClickHandlers() {
168         for (int y = 0; y < circles.length; y++) {
169             for (int x = 0; x < circles[0].length; x++) {
170                 final int cX = x;
171                 final int cY = y;
172                 circles[y][x].addEventHandler(MouseEvent.MOUSE_CLICKED, (MouseEvent event) -> {
173                     if (!game.getAllowedToRotate()) {
174                         setMarble(cX, cY);
175                     }
176                 });
177             }
178         }
179     }
180 
181     private void fillCircles() {
182         Marble[][] mArr = game.getBoard().toMarbleArray();
183         for (int y = 0; y < mArr.length; y++) {
184             for (int x = 0; x < mArr[0].length; x++) {
185                 if (mArr[y][x] != null) {
186                     switch (mArr[y][x].getSymbol()) {
187                         case X:
188                             circles[y][x].setStyle("-fx-fill:#000000");
189                             break;
190                         case O:
191                             circles[y][x].setStyle("-fx-fill:#FFFFFF");
192                             break;
193                     }
194                 } else {
195                     circles[y][x].setStyle("-fx-fill:rgba(0,0,0,0)");
196                 }
197             }
198         }
199     }

209     private void readyToSetMarble() {
210         baseScene.getRoot().lookup("AnchorPane").getStyleClass().add("readyToSetMarble");
211         baseScene.getRoot().lookup("AnchorPane").getStyleClass().remove("readyToRotate");
212         fillCircles();
213         for (Node n : rotateButtons) {
214             n.setVisible(false);
215         }
216         helpLabel.setText(game.whoseTurn() + " - Click where you'd want to put the marble");
217     }
218 
219     private void readyToRotate() {
220         baseScene.getRoot().lookup("AnchorPane").getStyleClass().add("readyToRotate");
221         baseScene.getRoot().lookup("AnchorPane").getStyleClass().remove("readyToSetMarble");
222         fillCircles();
223         for (Node n : rotateButtons) {
224             int x = game.getBoard().getLastRotatedTileX();
225             int y = game.getBoard().getLastRotatedTileY();
226             Direction d = game.getBoard().getLastRotatedTileDirection() != null ? game.getBoard().getLastRotatedTileDirection().opposite : null;
227             String id = d != null ? (d == Direction.CLOCKWISE ? "clockwise" : "counterClockwise") + x + y : null;
228             n.setVisible(true);
229             if (game.getBoard().getLastRotatedTile() != null && n.getId().equals(id)) {
230                 n.setVisible(false);
231             }
232         }
233         helpLabel.setText(game.whoseTurn() + " - Click the button on the tile you want to rotate");
234     }

236     private void setMarble(int x, int y) {
237         try {
238             game.addMarble(x, y);
239             readyToRotate();
240         } catch (Exception e) {
241         }
242     }

244     private void rotateTile(int x, int y, Direction d) {
245         try {
246             game.rotateTile(x, y, d);
247             errorLabel.setText("");
248             Line line = game.checkLines();
249             if (line != null) {
250                 winGame(line);
251             } else if (game.isEven()) {
252                 evenGame();
253             } else {
254                 nextTurn();
255             }
256         } catch (Exception e) {
257             if (e instanceof PentagoGameRuleException) {
258                 errorLabel.setText("Illegal direction!");
259             }
260         }
261     }
262 
263     private void nextTurn() {
264         if (game.whoseTurn() instanceof Bot) {
265             lockBoardForBot();
266             game.whoseTurn().makeMove();
267             nextTurn();
268         } else {
269             unlockBoardForPlayer();
270             readyToSetMarble();
271         }
272     }

286     private void winGame(Line line) throws IOException {
287         replaceSceneContent("fxml/GameOver.fxml");
288 
289         initTiles();
290         initCircles();
291         fillCircles();
292 
293         for (Integer[] coords : line.getCoordinates()) {
294             int x = coords[0];
295             int y = coords[1];
296             circles[y][x].strokeWidthProperty().setValue(5);
297             circles[y][x].strokeProperty().setValue(Paint.valueOf("#ffd700"));
298             String color;
299             if (line.getSymbol().equals(Symbol.O)) {
300                 color = "#FFFFFF";
301             } else {
302                 color = "#000000";
303             }
304             circles[y][x].setFill(Paint.valueOf(color));
305         }
306 
307         Text text = new Text(line.getPlayer() + " won!");
308         text.setFont(new Font("Arial", 72));
309         text.setFill(Paint.valueOf("#FF0000"));
310         TextFlow textFlow = (TextFlow) (baseScene.getRoot().lookup("#winText"));
311         textFlow.setVisible(true);
312         textFlow.getChildren().add(text);
313         initMainMenu();
314     }

330     /**
331      * Changes the view to a fxml view by the parameter 'fxml'.
332      *
333      * @param fxml the url of the fxml file
334      * @return the page created
335      * @throws IOException if the fxml file is not found
336      */
337     private void replaceSceneContent(String fxml) throws IOException {
338         Pane page = (Pane) FXMLLoader.load(getClass().getClassLoader().getResource(fxml));
339         baseScene = primaryStage.getScene();
340         if (baseScene == null) {
341             baseScene = new Scene(page);
342             primaryStage.setScene(baseScene);
343         } else {
344             baseScene.setRoot(page);
345         }
346         primaryStage.sizeToScene();
347     }
